import time
import asyncio
from typing import Dict, List, Any, Optional, Tuple


class NestedHierarchyTracker:
    """
    Tracks agent/tool timing with correct parent-child relationships for
    sequential, parallel, loop, and AgentTool execution patterns.

    Lead's version adopted wholesale with one fix
    ----------------------------------------------
    The lead replaced the previous (inv_id, author, call_id) exact-match dict
    with a call_id-only FIFO list, reasoning that responses may arrive
    out-of-order and the same call_id can appear across iterations.

    However the FIFO approach is ambiguous when two parallel agents in the
    same invocation both issue a call with the same call_id — the FIFO list
    contains both tool_keys and always pops the first one regardless of
    which agent sent the response event.  ADK guarantees that a
    function_response event carries the same author as the function_call
    that produced it, so (inv_id, author, call_id) is always unambiguous
    and covers the out-of-order / loop-reuse cases too.

    Everything else in the lead's version is correct and kept:
      - FIFO structure replaced with Dict[(inv_id, author, call_id) -> tool_key]
      - _last_traj_index cleared on close (correct: prevents stale attachment)
      - `if ref` on (i, j) tuple is fine — non-empty tuple is always truthy
      - any() instead of list comprehension in _determine_parent_and_scope
      - setdefault / generator patterns kept
    """

    def __init__(self):
        self.trajectory: List[Dict[str, Any]] = []
        self.active_spans: Dict[str, Dict[str, Any]] = {}
        self._loop_counts: Dict[str, int] = {}
        self._context_stack: List[str] = []
        self._scope_creator_sks: set = set()
        self._pending_agent_calls: Dict[Tuple[str, str], str] = {}
        self._sk_to_agent_key: Dict[str, str] = {}
        self._span_index: Dict[str, int] = {}
        self._tool_index: Dict[Tuple[str, str], Tuple[int, int]] = {}

        # (inv_id, author, call_id) -> tool_key
        # Three-part key is unambiguous even when parallel agents share call_id,
        # because ADK delivers function_response on the same author that issued
        # the function_call — so author+call_id together identify one in-flight tool.
        self._call_id_to_tool_key: Dict[Tuple[str, str, str], str] = {}

        self._last_traj_index: Dict[str, int] = {}
        self._children_of: Dict[str, set] = {}
        self._lock = asyncio.Lock()

    # ── Public API ─────────────────────────────────────────────────────────

    async def process_event(self, event) -> None:
        async with self._lock:
            self._handle_event(event)

    def finalize(self) -> List[Dict[str, Any]]:
        end_perf = time.perf_counter()
        for key in list(self.active_spans.keys()):
            data = self.active_spans.pop(key)
            if data["type"] == "tool":
                self._set_tool_duration(
                    data["span_key"], data["id"],
                    round(end_perf - data["start_perf"], 3),
                )
                self._call_id_to_tool_key.pop(
                    (data["inv_id"], data["author"], data["id"]), None
                )
            else:
                self._set_agent_duration(
                    data["span_key"],
                    round(end_perf - data["start_perf"], 3),
                )
                self._cleanup_agent(data["span_key"])
        return self.trajectory

    # ── Core event handler ─────────────────────────────────────────────────

    def _handle_event(self, event) -> None:
        curr_perf = time.perf_counter()
        curr_wall = time.time()
        inv_id    = event.get("invocation_id")
        author    = event.get("author")
        if not inv_id or not author:
            return

        agent_key = f"{inv_id}::{author}"
        is_final  = bool(event.get("is_final"))

        # ── Step 1: CLOSE ──────────────────────────────────────────────────
        if author != "user" and is_final:
            data = self.active_spans.pop(agent_key, None)
            if data:
                sk = data["span_key"]
                self._set_agent_duration(sk, round(curr_perf - data["start_perf"], 3))
                self._cleanup_agent(sk)

        # ── Step 2: OPEN ───────────────────────────────────────────────────
        has_calls   = bool(event.get("function_calls"))
        is_new_start = (
            author != "user"
            and agent_key not in self.active_spans
            and (not is_final or has_calls)
        )

        if is_new_start:
            iteration = self._loop_counts.get(agent_key, 0)
            self._loop_counts[agent_key] = iteration + 1
            sk = f"{inv_id}::{author}::{iteration}"

            parent_sk, is_scope = self._determine_parent_and_scope(author, inv_id)

            self.active_spans[agent_key] = {
                "type": "agent", "start_perf": curr_perf, "span_key": sk,
            }
            self._context_stack.append(sk)
            self._sk_to_agent_key[sk] = agent_key

            if is_scope:
                self._scope_creator_sks.add(sk)
                self._children_of[sk] = set()

            if parent_sk:
                self._children_of.setdefault(parent_sk, set()).add(sk)

            level = sum(1 for s in self._context_stack if s in self._scope_creator_sks)
            if is_scope:
                level -= 1

            label = author if iteration == 0 else f"{author}[loop {iteration}]"
            entry = {
                "type": "agent", "span_key": sk, "inv_id": inv_id, "author": author,
                "parent_key": parent_sk, "name": label, "start_time": curr_wall,
                "duration": None, "tools": [], "level": level,
            }
            idx = len(self.trajectory)
            self._span_index[sk]             = idx
            self._last_traj_index[agent_key] = idx
            self.trajectory.append(entry)

        # ── Step 3: TOOL CALLS ─────────────────────────────────────────────
        for call in event.get("function_calls", []):
            agent_data = self.active_spans.get(agent_key)
            current_sk = agent_data["span_key"] if agent_data else f"{agent_key}::?"
            tool_key   = f"tool::{current_sk}::{call['id']}"

            self.active_spans[tool_key] = {
                "type": "tool", "start_perf": curr_perf,
                "id": call["id"], "span_key": current_sk,
                "inv_id": inv_id, "author": author,
            }
            self._call_id_to_tool_key[(inv_id, author, call["id"])] = tool_key

            traj_idx = self._last_traj_index.get(agent_key)
            if traj_idx is not None:
                entry = self.trajectory[traj_idx]
                entry["tools"].append({
                    "id": call["id"], "name": call["name"],
                    "start_time": curr_wall, "duration": None,
                })
                self._tool_index[(current_sk, call["id"])] = (
                    traj_idx, len(entry["tools"]) - 1,
                )

            if agent_data:
                self._pending_agent_calls[(inv_id, call["name"])] = agent_data["span_key"]

        # ── Step 4: TOOL RESPONSES ─────────────────────────────────────────
        for resp in event.get("function_responses", []):
            tool_key = self._call_id_to_tool_key.pop((inv_id, author, resp["id"]), None)
            if tool_key:
                data = self.active_spans.pop(tool_key, None)
                if data:
                    self._set_tool_duration(
                        data["span_key"], resp["id"],
                        round(curr_perf - data["start_perf"], 3),
                    )

    # ── Parent / scope logic ───────────────────────────────────────────────

    def _determine_parent_and_scope(
        self, author: str, inv_id: str
    ) -> Tuple[Optional[str], bool]:
        key = (inv_id, author)
        if key in self._pending_agent_calls:
            return self._pending_agent_calls.pop(key), True

        if not any(v["type"] == "agent" for v in self.active_spans.values()):
            return None, True

        for candidate in reversed(self._context_stack):
            if candidate in self._scope_creator_sks:
                return candidate, False

        closed_scopes = self._scope_creator_sks - set(self._context_stack)
        if closed_scopes:
            best = max(closed_scopes, key=lambda sk: self._span_index.get(sk, -1))
            return best, False

        return None, False

    # ── Duration setters ──────────────────────────────────────────────────

    def _set_agent_duration(self, span_key: str, duration: float) -> None:
        idx = self._span_index.get(span_key)
        if idx is not None:
            self.trajectory[idx]["duration"] = duration

    def _set_tool_duration(self, span_key: str, call_id: str, duration: float) -> None:
        ref = self._tool_index.pop((span_key, call_id), None)
        if ref:
            i, j = ref
            self.trajectory[i]["tools"][j]["duration"] = duration

    # ── Agent cleanup ─────────────────────────────────────────────────────

    def _cleanup_agent(self, sk: str) -> None:
        try:
            self._context_stack.remove(sk)
        except ValueError:
            pass

        agent_key = self._sk_to_agent_key.get(sk)
        if agent_key:
            self._last_traj_index.pop(agent_key, None)

        if sk in self._scope_creator_sks:
            active_sks = {
                v["span_key"] for v in self.active_spans.values()
                if v["type"] == "agent"
            }
            if not (self._children_of.get(sk, set()) & active_sks):
                self._scope_creator_sks.discard(sk)
                self._children_of.pop(sk, None)

        for parent_sk, child_set in list(self._children_of.items()):
            if sk in child_set:
                child_set.discard(sk)
                parent_agent_key = self._sk_to_agent_key.get(parent_sk)
                parent_closed = (
                    not parent_agent_key
                    or parent_agent_key not in self.active_spans
                )
                if (parent_sk in self._scope_creator_sks
                        and parent_closed and not child_set):
                    self._scope_creator_sks.discard(parent_sk)
                    self._children_of.pop(parent_sk, None)
                break

        self._sk_to_agent_key.pop(sk, None)

    # ── Pretty print ──────────────────────────────────────────────────────

    def print_summary(self) -> None:
        for item in self.trajectory:
            indent = "  " * item.get("level", 0)
            dur    = f"{item['duration']}s" if item["duration"] is not None else "n/a"
            print(f"{indent}Agent: {item['name']}  dur={dur}")
            for t in item.get("tools", []):
                tdur = f"{t['duration']}s" if t["duration"] is not None else "n/a"
                print(f"{indent}  └── Tool: {t['name']}  dur={tdur}")
